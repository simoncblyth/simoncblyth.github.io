In [3]: wl = np.linspace(100.,730.,10, dtype=np.float32)

In [4]: ri = mat.refractive_index(wl)

In [8]: al = mat.absorption_length(wl)

In [9]: sl = mat.scattering_length(wl)

In [10]: rp = mat.reemission_prob(wl)

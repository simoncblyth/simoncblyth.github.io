// For capability <2.0 (but >=1.3) GPUs
 #define ATOMIC_ADD(x,y) MyAtomicAdd (x, (float)(y))
 #else
 #define ATOMIC_ADD(x,y) atomicAdd (x, (float)(y))
 #endif
